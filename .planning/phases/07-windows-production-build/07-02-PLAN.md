Modify the `normalizeAudioLocation()` function in renderer/App.jsx. The function currently always returns `file://` URLs. In production on Windows, it must return `rbfa-audio://` URLs instead.

Find the function (starts around line 260). Replace the final URL construction section (after URI decoding, where it builds `fileUrl`) with logic that checks for production mode:

```javascript
  // Read isProduction from the bridge (added in Plan 01)
  const isProduction = Boolean(bridge?.isProduction);

  // In production, use custom protocol to bypass webSecurity restrictions
  if (isProduction) {
    // rbfa-audio://local/<encoded-path>
    // The path must be a native filesystem path (C:/Users/... on Windows)
    // The protocol handler in main.js will decode and serve the file
    let nativePath = cleaned;

    // On Windows production, paths should be C:/... format (not /mnt/...)
    // The cleaned path at this point is already in the correct format
    // because we only do WSL /mnt/ mapping when platform is linux

    const encodedPath = encodeURIComponent(nativePath);
    const audioUrl = `rbfa-audio://local/${encodedPath}`;

    if (options.debug) {
      console.log('[rbfa] normalizeAudioLocation RESULT (production)', {
        fsPath: cleaned,
        audioUrl,
        platform,
        isProduction
      });
    }

    return audioUrl;
  }

  // Dev mode: use file:// URLs as before
  const encoded = encodeURI(cleaned)
    .replace(/#/g, '%23')
    .replace(/\?/g, '%3F');

  let fileUrl = '';
  if (/^[A-Za-z]:\//.test(cleaned)) {
    fileUrl = `file:///${encoded}`;
  } else if (cleaned.startsWith('/')) {
    fileUrl = `file://${encoded}`;
  } else {
    const hasScheme = /^[A-Za-z][A-Za-z0-9+.-]*:/.test(encoded);
    if (!hasScheme) {
      if (/^[A-Za-z]:\//.test(encoded)) {
        fileUrl = `file:///${encoded}`;
      } else if (encoded.startsWith('/')) {
        fileUrl = `file://${encoded}`;
      } else {
        fileUrl = encoded;
      }
    } else {
      fileUrl = encoded;
    }
  }
```

Keep the existing debug logging at the end for the dev path.

IMPORTANT: Do NOT change the cleaning/decoding logic earlier in the function. Only modify the URL construction at the end.

IMPORTANT: The `isProduction` variable must be read from the bridge BEFORE the URL construction section. Add it near where `platform` and `isWsl` are read at the top of the function.

**Part A: Auto-updater in main.js**

1. Add import at top of electron/main.js:
```javascript
import { autoUpdater } from 'electron-updater';
```

2. Inside the `app.whenReady().then(() => { ... })` block, AFTER `createWindow()`, add:

```javascript
  // Auto-update: check for updates in packaged builds only
  if (app.isPackaged) {
    autoUpdater.logger = {
      info: (...args) => console.log('[rbfa:updater]', ...args),
      warn: (...args) => console.warn('[rbfa:updater]', ...args),
      error: (...args) => console.error('[rbfa:updater]', ...args),
      debug: (...args) => console.log('[rbfa:updater:debug]', ...args)
    };

    autoUpdater.checkForUpdatesAndNotify().catch((err) => {
      console.error('[rbfa:updater] Update check failed:', err.message);
    });
  }
```

Do NOT add periodic update checks, dialog boxes, or quitAndInstall logic. Keep it simple: check on launch, notify if update available. More sophisticated update UX deferred to Phase 8.

Do NOT wrap in try/catch that swallows errors silently. The .catch() on the promise is sufficient.

**Part B: Create placeholder app icon**

Create `build/icon.png` as a 256x256 PNG. Use Node.js to generate a minimal valid PNG programmatically:

```bash
node -e "
const fs = require('fs');
// Generate a minimal 256x256 PNG with a solid dark background
// Using raw PNG construction for a simple colored square
const width = 256, height = 256;

// PNG signature
const signature = Buffer.from([137, 80, 78, 71, 13, 10, 26, 10]);

// IHDR chunk
const ihdr_data = Buffer.alloc(13);
ihdr_data.writeUInt32BE(width, 0);
ihdr_data.writeUInt32BE(height, 4);
ihdr_data[8] = 8;  // bit depth
ihdr_data[9] = 2;  // color type (RGB)
ihdr_data[10] = 0; // compression
ihdr_data[11] = 0; // filter
ihdr_data[12] = 0; // interlace

function makeChunk(type, data) {
  const len = Buffer.alloc(4);
  len.writeUInt32BE(data.length, 0);
  const typeBuffer = Buffer.from(type);
  const crc32 = require('zlib').crc32 || (() => {
    // Simple CRC32 fallback
    let table = new Uint32Array(256);
    for (let i = 0; i < 256; i++) {
      let c = i;
      for (let j = 0; j < 8; j++) c = c & 1 ? 0xEDB88320 ^ (c >>> 1) : c >>> 1;
      table[i] = c;
    }
    return (buf) => {
      let crc = 0xFFFFFFFF;
      const combined = Buffer.concat([typeBuffer, buf]);
      for (let i = 0; i < combined.length; i++) crc = table[(crc ^ combined[i]) & 0xFF] ^ (crc >>> 8);
      return (crc ^ 0xFFFFFFFF) >>> 0;
    };
  })();
  const crcVal = Buffer.alloc(4);
  const crcInput = Buffer.concat([typeBuffer, data]);
  // Use zlib crc32 if available
  let c;
  try { c = require('zlib').crc32(crcInput); } catch { c = 0; }
  crcVal.writeUInt32BE(c >>> 0, 0);
  return Buffer.concat([len, typeBuffer, data, crcVal]);
}

// Build raw image data (filter byte + RGB for each row)
const raw = Buffer.alloc(height * (1 + width * 3));
for (let y = 0; y < height; y++) {
  const rowOffset = y * (1 + width * 3);
  raw[rowOffset] = 0; // no filter
  for (let x = 0; x < width; x++) {
    const px = rowOffset + 1 + x * 3;
    raw[px] = 30;    // R - dark blue-gray
    raw[px+1] = 35;  // G
    raw[px+2] = 50;  // B
  }
}

const zlib = require('zlib');
const compressed = zlib.deflateSync(raw);

const ihdrChunk = makeChunk('IHDR', ihdr_data);
const idatChunk = makeChunk('IDAT', compressed);
const iendChunk = makeChunk('IEND', Buffer.alloc(0));

const png = Buffer.concat([signature, ihdrChunk, idatChunk, iendChunk]);
fs.writeFileSync('build/icon.png', png);
console.log('Created build/icon.png (' + png.length + ' bytes)');
"
```
